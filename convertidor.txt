    public void con(){
         switch (d){
             case "MXN":
             if (a.equals("GTQ")){
                conver = monto*0.38;
             }
             else if (a.equals("USD")){
                conver = monto*0.049;
             }
             else if(a.equals("EUR")){
                conver = monto*0.047;}
             else {
                conver = monto;
             }
             break;
             
             case "USD":
             if (a.equals("GTQ")){
                conver = monto*7.67;
             }
             else if (a.equals("MXN")){
                conver = monto*20.43;
             }
             else if(a.equals("EUR")){
                conver = monto*0.95;}
             else {
                conver = monto;
             }
             break;
             
             case "GTQ":
             if (a.equals("MXN")){
                conver = monto*2.67;
             }
             else if (a.equals("USD")){
                conver = monto*0.13;
             }
             else if(a.equals("EUR")){
                conver = monto*0.12;}
             else {
                conver = monto;
             }
             break;
               
         } 
         lr.setText(Mon(monto,d)+"  "+Mon(conver,a));
    }
